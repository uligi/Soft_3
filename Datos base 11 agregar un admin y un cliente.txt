


USE DUNAMIS_SA;
GO

-- Agregar un administrador usando el procedimiento almacenado CrearUsuario
EXEC CrearUsuario 
    @Cedula = 12345678, 
    @Nombre = 'Admin', 
    @Apellido1 = 'Administrador', 
    @Apellido2 = 'Sistema', 
    @Direccion = 'Oficina Central', 
    @DistritoID = 1, 
    @CantonID = 1, 
    @ProvinciaID = 1, 
    @Telefono = '1111-1111', 
    @TipoTelefonoID = 3, 
    @Correo = 'admin@sistema.com', 
    @TipoCorreoID = 2, 
    @RolID = 1, -- ID para el rol de Administrador
    @Contrasena = 'admin1234';
GO
